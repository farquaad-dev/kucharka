\subsection{ Vajíčková nátierka }

\begin{ingredient}
  
  \def\portions{  }
  \textbf{ {\normalsize Ingrediencie (4 porcie):} }

  \begin{main}
      \item 4ks vajíčko
      \item 125 g maslo
      \item 1 čl horčica
      \item pečivo
  \end{main}
  
\end{ingredient}
\begin{recipe}
\textbf{ {\normalsize Príprava:} }
\begin{enumerate}

  \item{Vajíčka uvaríme na tvrdo - 8-10 minút}
  \item{Olúpané vajíčka zmiešame s maslom a horčicou, osolíme}

\end{enumerate}
\end{recipe}
