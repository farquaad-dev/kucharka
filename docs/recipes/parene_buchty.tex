\subsection{Buchty na pare}
\PrepTime{45}
\CookingTime{10}
\CookingTempe{180}
\TypeCooking{Plaque}
\NbPerson{4}
\Image{0 0 270 270}{images/buchty_na_pare} %style 2
%------------------------------------------

\begin{ingredient}
%\vspace{0.5cm}
\begin{main}
	\item 700g polohrubá múka
	\item 1ks droždie
	\item 400ml mlieko
	\item 30g kryštálový cukor
	\item štipka soľ
	\item plnka (nutela, džem)
\end{main}
\end{ingredient}%no space with \begin{recipe}
\begin{recipe}


\step{Z teplého mlieka, droždia, cukru, kúsku múky a štipky soli urobíme kvások}
\step{Vypracujeme hladké cesto, necháme 45 minút vykysnúť}
\step{Vyvaľkáme na 1cm a urobíme buchty}
\step{Paríme asi 10 minút}	

\end{recipe}
