\subsection{Lasagne s brokolicou}

\begin{ingredient}
%\vspace{0.5cm}
\begin{main}
	\item lasagnové pláty
	\item brokolica
	\item tvrdý syr
\end{main}
\begin{subingredient}{Bešamel}
	\item 50g maslo
	\item štipka soli
	\item 50g hladkej múky
	\item 500ml mlieka
	\item nejaké korenie
\end{subingredient}
\end{ingredient}%no space with \begin{recipe}
\begin{recipe}


\step{Pripravíme si bešamel}
\substep[Bešamel]{Roztopiť maslo, pridať múku}
\substep{Pridávať mlieko}
\substep{Osoliť, okoreniť}

\step{Dáme do taniera vriacu vodu, postupne do nej namáčame pláty a ukladáme do zapekacej misy}
\step{Na vrstvu plátov dáme vrstvu bešamelu a brokolice (poprípade šunky)}
\step{Na poslednú vrstvu nedávame brokolicu, ale poukladáme/postrúhame tvrdý syr}

\end{recipe}
