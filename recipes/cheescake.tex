\begin{recipe}

\step{Maslové sušienky rozdrviť a primiešať maslo}
\step{Vystelieme formu na pečenie papierom a natlačíme na kraje sušienky s maslom}
\step{Dáme piecť na 15 minút korpus a zatiaľ zmiešame mascarpone s vajíčkami, cukrom, smotanou}	
\step{Zmesou naplníme korpus a dáme piexť na 40-50 minút}
\substep[Slaný karamel]{Skaramelizujeme cukor}
\substep{Pridáme maslo}
\substep{Prilejeme smotanu}
\substep{Prídáme 1/2 lyžičky soli}
\step{Polejeme, necháme v chladničke vychladnúť}

\end{recipe}
