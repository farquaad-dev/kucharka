\begin{main}
\item
\end{main}
\begin{subingredient}{Pláty}
	\item 2 balenia hotových plátov, alebo
	\item 250g hladká múka
  \item 250g polohrubá múka
  \item 150g cukor
  \item 80g maslo
  \item 3ks vajíčko
  \item 1 štipka sódy bikarbóny
\end{subingredient}
\begin{subingredient}{Plnka}
	\item 500ml mlieko
	\item 2ks Zlatý klas
	\item 250g maslo
	\item 100g cukor
  \item 100g jahodnový džem
\end{subingredient}
\begin{subingredient}{Poleva}
	\item 100g práškový cukor
	\item 100g maslo
  \item 2ks vajíčko
  \item 1PL kakao
\end{subingredient}
\end{ingredient}%no space with \begin{recipe}
\begin{recipe}


\step{Pripravíme pláty}
\substep[Pláty]{Nad parou zmiešame cukor, maslo, vajíčka, med, škoricu a sódu bikarbónu.}
\substep{Za stáleho miešania chvíľu varíme. Odstavíme a primiešame hladkú aj polohrubú múku. }
\substep{Z hmoty vypracujeme cesto, ktoré si rozdelíme na 4 rovnaké časti.}
\substep{Rozvaľkáme a po jednom pečieme každý plát približne 6 minút pri 180 stupňoch. }


\step{Popri pečení pripravíme plnku:}
\substep[Plnka]{Uvaríme Zlatý klas a necháme h vychladnúť.}
\substep{Po vychladnutí primiešame maslo a cukor.}

\step{Na prvé dva pláty natrieme 2/3 plnky.}
\step{Tretí potrieme džemom a na džem natrieme zvyšok plnky.}
\step{Posledný plát položíme na tretí.}

\step{Pripravíme polevu}
\substep{Nad parou rozšľaháme vajíčka,cukor a kakao.}
\substep{Do uvarenej horúcej zmesi vmiešame maslo.}

\step{Naplnené pláty polejeme polevou a posypeme nasekanými orieškami.}


\end{recipe}
