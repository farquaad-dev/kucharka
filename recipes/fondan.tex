\begin{ingredient}
%\vspace{0.5cm}
\begin{main}
	\item 200g marshmallow
	\item 1 ČL maslo
	\item 2 štipky kyseliny citrónovej
	\item 250g práškový cukor
\end{main}
\end{ingredient}%no space with \begin{recipe}
\begin{recipe}

\step{Marshmallow, maslo a citrodeko dáme do misky a na minútu - dve do mikrovlnky. (Každých 15s premiešať.)}
\step{Pridávame preosiaty cukor. (Kým sa dá, miešame v miske)}
\step{Vyklopíme na dosku na preosiaty cukor, vyrobíme tuhšie cesto.}	
\step{Vyvaľkať a preniesť valčekom na tortu}

\end{recipe}
